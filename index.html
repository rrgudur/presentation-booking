Google Sheets Integration

Bookings and session settings are sent to Google Sheets via a Apps Script Web App.

One-time setup

  1. Open Google Sheets → create a new spreadsheet.
  2. Go to Extensions → Apps Script.
  3. Delete any existing code and paste the script below.
  4. Click Deploy → New deployment → Web App.
  5. Set Execute as = Me, Who has access = Anyone.
  6. Copy the Web App URL and paste it in the field below.
  7. Click Save & Sync to Sheets.
▶ Show Apps Script code 
function doPost(e) {
  try {
    var ss   = SpreadsheetApp.getActiveSpreadsheet();
    var raw  = (e && e.postData && e.postData.contents) ? e.postData.contents : '{}';
    var data = JSON.parse(raw);

    // ── SESSION SETTINGS ──────────────────────────────────
    if (data.type === 'settings') {
      var sSheet = ss.getSheetByName('Session Settings');
      if (!sSheet) {
        sSheet = ss.insertSheet('Session Settings');
        var hdr = ['Saved At','Session Date','Course','Start Time',
                   'End Time','Break Start','Break End'];
        sSheet.appendRow(hdr);
        sSheet.getRange(1, 1, 1, hdr.length)
              .setFontWeight('bold')
              .setBackground('#0891b2')
              .setFontColor('#ffffff');
      }
      sSheet.appendRow([
        data.savedAt,
        data.date       || '',
        data.course     || '',
        data.startTime  || '',
        data.endTime    || '',
        data.breakStart || '',
        data.breakEnd   || ''
      ]);
      return ContentService
        .createTextOutput(JSON.stringify({status:'ok', type:'settings'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // ── BOOKING ───────────────────────────────────────────
    var sheet = ss.getSheetByName('Bookings') || ss.getActiveSheet();
    if (sheet.getLastRow() === 0) {
      var bHdr = ['Booking #','Time Slot','Full Name','Student ID',
                  'Email','Project Title','Description','Booked At'];
      sheet.appendRow(bHdr);
      sheet.getRange(1, 1, 1, bHdr.length)
           .setFontWeight('bold')
           .setBackground('#2a5fb0')
           .setFontColor('#ffffff');
    }
    var rowNum = sheet.getLastRow();
    sheet.appendRow([
      rowNum,
      data.timeSlot,
      data.name,
      data.studentId,
      data.email       || '',
      data.title,
      data.description,
      data.bookedAt
    ]);
    return ContentService
      .createTextOutput(JSON.stringify({status:'ok', row:rowNum}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error', message:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok', message:'PresentSlot API ready'}))
    .setMimeType(ContentService.MimeType.JSON);
}
