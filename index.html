Google Sheets Integration

Bookings are saved to Google Sheets via a Google Apps Script Web App.

How to set up Google Sheets

  1. Open Google Sheets → create a new spreadsheet.
  2. Go to Extensions → Apps Script.
  3. Delete existing code and paste the script below.
  4. Click Deploy → New deployment → Web App.
  5. Set "Execute as" = Me and "Who has access" = Anyone.
  6. Copy the Web App URL and paste it below.
Show Apps Script code 
function doPost(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var data = JSON.parse(e.postData.contents);

    // ── SETTINGS record ──
    if (data.type === 'settings') {
      var sSheet = ss.getSheetByName('Session Settings');
      if (!sSheet) {
        sSheet = ss.insertSheet('Session Settings');
        sSheet.appendRow(['Saved At','Session Date','Course','Start Time','End Time','Break Start','Break End']);
        sSheet.getRange(1,1,1,7).setFontWeight('bold')
              .setBackground('#0891b2').setFontColor('#ffffff');
      }
      sSheet.appendRow([
        data.savedAt,
        data.date || '',
        data.course || '',
        data.startTime || '',
        data.endTime || '',
        data.breakStart || '',
        data.breakEnd || ''
      ]);
      return ContentService
        .createTextOutput(JSON.stringify({status:'ok',type:'settings'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // ── BOOKING record (default) ──
    var sheet = ss.getSheetByName('Bookings') || ss.getActiveSheet();
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(['Booking #','Time Slot','Full Name','Student ID',
                       'Email','Project Title','Description','Booked At']);
      sheet.getRange(1,1,1,8).setFontWeight('bold')
           .setBackground('#2a5fb0').setFontColor('#ffffff');
    }
    var rowNum = sheet.getLastRow();
    sheet.appendRow([rowNum, data.timeSlot, data.name, data.studentId,
                     data.email||'', data.title, data.description, data.bookedAt]);
    return ContentService
      .createTextOutput(JSON.stringify({status:'ok',row:rowNum}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error',message:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e) {
